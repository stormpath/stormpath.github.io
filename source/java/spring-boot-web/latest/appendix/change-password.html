<html xmlns:th="http://www.thymeleaf.org">
<head>
    <title th:text="#{stormpath.web.changePassword.title}">|Change Your Password|</title>
    <!--/*/ <th:block th:include="${headViewName} :: ${headFragmentSelector}"/> /*/-->
</head>
<body class="login">

    <div class="container custom-container">

        <div class="va-wrapper">

            <div class="view login-view container">

                <div class="box row">

                    <div class="email-password-area col-xs-12 large col-sm-12">

                        <div class="header">
                            <span th:text="#{stormpath.web.changePassword.form.title}">|Change Your Password|</span>
                            <p th:text="#{stormpath.web.changePassword.form.instructions}">|Enter your new account password below.
                                Once confirmed, you'll be logged into your account and your new password will be active.|</p>
                        </div>

                        <div class="alert alert-dismissable alert-danger bad-login" th:if="${errors != null and not #lists.isEmpty(errors)}">
                            <button type="button" class="close" data-dismiss="alert">&times;</button>
                            <!--/*/ <th-block data-th-each="error : ${errors}"> /*/-->
                            <p th:text="${error.message}">|Error message here.|</p>
                            <!--/*/ </th-block> /*/-->
                        </div>

                        <form method="post" role="form" class="login-form form-horizontal">

                            <div th:each="field : ${form.fields}" form-group="true" th:class="'form-group group-' + ${field.name}">

                                <th:block th:if="${field.type == 'hidden'}">
                                    <input th:name="${field.name}" th:type="${field.type}" th:value="${field.value}"/>
                                </th:block>
                                <th:block th:if="${field.type != 'hidden'}">
                                    <label class="col-sm-4" th:text="${field.label}">|Field label|</label>
                                    <div class="col-sm-8">
                                        <input th:name="${field.name}" th:value="${field.value}" th:type="${field.type}"
                                               th:placeholder="${field.placeholder}"
                                               th:required="${field.required}" class="form-control"/>
                                    </div>
                                </th:block>
                            </div>

                            <div>
                                <button type="submit" class="login btn btn-login btn-sp-green"
                                        th:text="#{stormpath.web.changePassword.form.button.value}">|Submit|</button>
                            </div>
                        </form>

                    </div>

                </div>

                <a th:href="@{${loginUri}}" class="to-login" th:text="#{stormpath.web.changePassword.form.loginLink.text}">|Back to Login|</a>

            </div>

        </div>

    </div>
</body>
</html>
